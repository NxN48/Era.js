第一章 第三节 加入一个页面

```python
import erajs.api as a # 导入erajs引擎的后端代码库
a.init() # 初始化引擎
def ui_start_game():
    a.page() # 新建一个页面
    a.h('开始游戏页面') # 显示标题文字
a.page() # 新建一个页面
a.h('这是Era.js教程') # 显示标题文字
a.t()
a.b('开始游戏', ui_start_game)
```
